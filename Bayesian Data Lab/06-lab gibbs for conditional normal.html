Exercise 3

# Load necessary libraries
library(ggplot2)

# Set initial values
X <- 0
Y <- 0
Z <- 0
n <- 1000  # Number of MCMC samples

# Initialize the matrix to store samples
samples_matrix <- matrix(nrow = n, ncol = 3)

# Gibbs sampler with matrix storage
for(i in 1:n){
  # Update X from its conditional distribution
  X <- rnorm(1, mean = Y*88/89+Z/99, sd = 94/495)  # Example conditional distribution
  
  # Update Y from its conditional distribution
  Y <- rnorm(1, mean = X*88/89+Z/99, sd = 94/495)  # Example conditional distribution
  
  # Update Z from its conditional distribution
  Z <- rnorm(1, mean = (X + Y) / 19, sd = 94/95)  # Example conditional distribution
  
  # Store the samples in the matrix
  samples_matrix[i, ] <- c(X, Y, Z)
}

# You can access samples_X, samples_Y, samples_Z like this:
samples_X <- samples_matrix[, 1]
samples_Y <- samples_matrix[, 2]
samples_Z <- samples_matrix[, 3]

# Load necessary libraries
library(ggplot2)
library(coda)
library(mvtnorm)

# Set initial values
X <- 0
Y <- 0
Z <- 0
n <- 1000  # Number of MCMC samples

sigmaXY<- matrix(c(0.99,0.89,0.89,0.99), ncol=2)

# Initialize the matrix to store samples
samples_matrix <- matrix(nrow = n, ncol = 3)

# Gibbs sampler with matrix storage
for(i in 1:n){
  # Update X Y from its conditional distribution
  XY <- rmvnorm(n=1, mean=c(0.1*Z,0.1*Z), sigma=sigmaXY)  # Example conditional distribution
  X <-XY[1]
  Y <-XY[2]
  
  
  # Update Z from its conditional distribution
  Z <- rnorm(1, mean = (X + Y) / 19, sd = 94/95)  # Example conditional distribution
  
  # Store the samples in the matrix
  samples_matrix[i, ] <- c(X, Y, Z)
}

# You can access samples_X, samples_Y, samples_Z like this:
samples_X <- samples_matrix[, 1]

samples_Z <- samples_matrix[, 3]

Compare two sampler

Second Gibbs Sampler (Blocked Updates): More efficient because it better accounts for dependencies among variables by updating correlated variables together. This approach can lead to faster convergence and more independent samples, making it a preferable choice for complex models with correlated parameters.
