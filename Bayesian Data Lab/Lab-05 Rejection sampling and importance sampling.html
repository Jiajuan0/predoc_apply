library(ggplot2)

# Define the target density function
f <- function(x) {
  sin(pi * x)^2
}

# Number of samples desired
nsim <- 10000

# Initialize variables for acceptance count
ac <- tc <- 0
res <- numeric(nsim)

# Rejection Sampling process, to generate nsim 
while (ac < nsim) {
  tc <- tc + 1
  x_proposed <- runif(1, 0, 1)
  u <- runif(1, 0, 1)

  if (u <= f(x_proposed)) {
    ac <- ac + 1
    res[ac] <- x_proposed
  }
}

#Accept Rate
print(ac/tc)
## [1] 0.51
# Plotting the density of the samples
ggplot(data.frame(x = res), aes(x)) +

Exercise 5: trying a different g: Beta(2,2)

library(ggplot2)


# Define the target density function
f <- function(x) {
  sin(pi * x)^2
}

# Number of samples desired
nsim <- 10000

# Find M
ss=seq(0.001,0.999,0.001)
M<-max(f(ss)/dbeta(ss,2,2))


# Initialize variables for acceptance count
ac <- tc <- 0
res <- numeric(nsim)
while (ac < nsim) {
  tc <- tc + 1
  x_proposed <- rbeta(1,2,2)
  u <- runif(1, 0, 1)

  if (u <= f(x_proposed)/(M*dbeta(x_proposed,2,2))) {#Try different M
    ac <- ac + 1
    res[ac] <- x_proposed
  }
}

#Accept Rate
print(ac/tc)
## [1] 0.75
# Plotting the density of the samples
ggplot(data.frame(x = res), aes(x)) +
  geom_histogram(aes(y = ..density..), bins = 50, fill = "blue", alpha = 0.4) +
  stat_function(fun = f, color = "red", size = 1.5) +
  labs(title = "Density of Samples and Function f(x) = sin(pi * x)^2",
       x = "x",
       y = "Density") +
  theme_minimal()

tumors <- read.csv(file = url("http://www.stat.columbia.edu/~gelman/book/data/rats.asc"),
                     skip = 2, header = T, sep = " ")[,c(1,2)]
y <- tumors$y
N <- tumors$N
n <- length(y)

stan_dat <- list(n = n, N = N, y =y, a = 1, b = 1)
fit_pool <- stan('lab-02-pool.stan', data = stan_dat, chains = 2, refresh = 0)
pool_output <- rstan::extract(fit_pool)
as <- c(1,10,25,100)
bs <- c(1,10,25,100)

# posterior parameters under the current Beta(1,1) prior
post_a0 <- 1 + sum(y)
post_b0 <- 1 + sum(N) - sum(y)

#sampling theta from q
theta_q=pool_output$theta

IS_sensi_mean <- matrix(0, length(as), length(bs))  # IS estimated posterior mean
exp_sensi_mean <- matrix(0, length(as), length(bs))  # theoretical posterior mean

for (i in (1:length(as))){
  for (j in (1:length(bs))){
    post_a1 <- i + sum(y)
    post_b1 <- j + sum(N) - sum(y)
    exp_sensi_mean[i,j]<-post_a1/(post_a1+post_b1)
    w=dbeta(theta_q,post_a1,post_b1)/dbeta(theta_q,post_a0,post_b0)
    IS_sensi_mean[i,j]<-sum(theta_q*w)/sum(w) #self-normalizing

  }
}

# plot the theoretical posterior means against the IS estimated posterior means
plot(c(IS_sensi_mean), c(exp_sensi_mean), xlab = "IS", ylab = "Expected")
